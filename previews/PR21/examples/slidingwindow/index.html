Sliding window analysis

using SimpleSDMLayers
using Plots
using Statistics

precipitation = worldclim(12; left=-80.0, right=-56.0, bottom=44.0, top=62.0)
SimpleSDMPredictor{Float32}(Union{Nothing, Float32}[896.0f0 856.0f0 … nothing nothing; 900.0f0 832.0f0 … nothing nothing; … ; 298.0f0 299.0f0 … nothing nothing; 295.0f0 300.0f0 … nothing nothing], -80.16666666666666, -56.0, 44.0, 61.83333333333333)
averaged = slidingwindow(precipitation, Statistics.mean, 100.0)
SimpleSDMPredictor{Any}(Any[904.0476f0 909.7857f0 … nothing nothing; 918.4167f0 923.34375f0 … nothing nothing; … ; 299.36365f0 299.36365f0 … nothing nothing; 299.36365f0 299.36365f0 … nothing nothing], -80.16666666666666, -56.0, 44.0, 61.83333333333333)
plot(precipitation, c=:alpine)
contour!(averaged, c=:white, lw=2.0)
