Sliding window analysis

using SimpleSDMLayers
using Plots
using Statistics

isothermality = worldclim(3; left=-80.0, right=-56.0, bottom=44.0, top=62.0)
SimpleSDMPredictor{Float32}(Union{Nothing, Float32}[25.963108f0 26.122736f0 … nothing nothing; 25.876137f0 26.508482f0 … nothing nothing; … ; 12.040338f0 11.811735f0 … nothing nothing; 12.147704f0 11.906481f0 … nothing nothing], -80.16666666666666, -56.0, 44.0, 61.83333333333333)
averaged = slidingwindow(isothermality, Statistics.mean, 100.0)

plot(averaged, fill=true, lw=0.0, c=:alpine)
