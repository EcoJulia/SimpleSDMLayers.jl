function get_most_common_landuse(f)
    f[isnan.(f)] .= 0.0
    sum(f) == 0 && return NaN
    return last(findmax(f))
end

function shannon(x)
    u = unique(x)
    filter!(!isnan, u)
    length(u) == 0 && return NaN
    c = [count(x.==v) for v in u]
    p = c./sum(c)
    return -sum(p.*log2.(p))
end
